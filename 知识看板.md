# 知识看板

> 知识网络全貌仪表盘。全动态计算——卡片元数据来自 Dataview 索引缓存，链接图由 `file.outlinks`/`file.inlinks` 实时构建，零文件 I/O，零预计算依赖。打开即当前状态。

```dataviewjs
// ═══════════════════════════════════════════════════════════
// 共享：链接图一次性构建，所有模块复用
// ═══════════════════════════════════════════════════════════

const cards = dv.pages('"02-知识卡片/知识卡片"');
const cardNames = new Set(cards.map(p => p.file.name));

// 构建 card→card 有向图（仅「相关链接」节，不含正文引用和来源引用）
const cardContents = await Promise.all(
    Array.from(cards).map(async (p) => {
        try {
            const content = await dv.io.load(p.file.path);
            return { name: p.file.name, content };
        } catch (_) {
            return { name: p.file.name, content: '' };
        }
    })
);

const outlinks = {};
const inlinks = {};
for (const { name, content } of cardContents) {
    // 全文扫描所有 wikilink（与 Python 网络分析脚本、知识检索 BFS 一致）
    const linkRegex = /\[\[([^\]]+)\]\]/g;
    const targets = [];
    let m;
    while ((m = linkRegex.exec(content)) !== null) {
        const t = m[1].split('|')[0].split('#')[0].trim().split('/').pop();
        if (cardNames.has(t) && t !== name) targets.push(t);
    }
    // 去重
    outlinks[name] = [...new Set(targets)];
    for (const t of targets) {
        if (!inlinks[t]) inlinks[t] = [];
        inlinks[t].push(name);
    }
}
const degree = (name) => (outlinks[name]?.length || 0) + (inlinks[name]?.length || 0);
const totalEdges = Object.values(outlinks).reduce((s, a) => s + a.length, 0);
const avgDegree = cardNames.size > 0 ? (2 * totalEdges / cardNames.size).toFixed(2) : '0';  // 度 = 入+出，总和 = 2E

// 死链检测：只检查 .md 链接（PDF/图片等非 md 文件 Dataview 不索引，必然误报，跳过）
const allPaths = new Set(dv.pages().map(p => p.file.path));
let deadLinks = 0;
for (const p of cards) {
    for (const l of (p.file.outlinks || [])) {
        if (!l.path.endsWith('.md')) continue;
        if (!allPaths.has(l.path)) deadLinks++;
    }
}

// 度分布直方图
const dist = { '<3': 0, '3-8': 0, '9-14': 0, '15+': 0 };
for (const name of cardNames) {
    const d = degree(name);
    if (d < 3) dist['<3']++;
    else if (d <= 8) dist['3-8']++;
    else if (d <= 14) dist['9-14']++;
    else dist['15+']++;
}
const lowLinkLt2 = dist['<3'];

// 双向链接
let biPairs = 0, totalPairs = 0;
for (const [src, targets] of Object.entries(outlinks)) {
    for (const tgt of targets) {
        totalPairs++;
        if (outlinks[tgt] && outlinks[tgt].includes(src)) biPairs++;
    }
}
const bidirectionalPct = totalPairs > 0 ? (biPairs * 100 / totalPairs).toFixed(1) : '0';

// 连通分量 BFS
const visitedCC = new Set();
let components = 0;
for (const name of cardNames) {
    if (visitedCC.has(name)) continue;
    components++;
    const queue = [name];
    visitedCC.add(name);
    while (queue.length) {
        const cur = queue.shift();
        const neighbors = new Set([...(outlinks[cur] || []), ...(inlinks[cur] || [])]);
        for (const nb of neighbors) {
            if (!visitedCC.has(nb)) { visitedCC.add(nb); queue.push(nb); }
        }
    }
}

// 关节点 Tarjan O(N+E)
let time = 0;
const disc = {}, low = {}, ap = new Set(), parent = {};
function dfs(u) {
    let children = 0;
    disc[u] = low[u] = ++time;
    const neighbors = new Set([...(outlinks[u] || []), ...(inlinks[u] || [])]);
    for (const v of neighbors) {
        if (!(v in disc)) {
            children++;
            parent[v] = u;
            dfs(v);
            low[u] = Math.min(low[u], low[v]);
            if (parent[u] === undefined && children > 1) ap.add(u);
            if (parent[u] !== undefined && low[v] >= disc[u]) ap.add(u);
        } else if (v !== parent[u]) {
            low[u] = Math.min(low[u], disc[v]);
        }
    }
}
for (const name of cardNames) {
    if (!(name in disc)) dfs(name);
}

// 网络直径 + 平均最短路径（全源 BFS，O(N·(N+E))，约 1100 次遍历）
const nodeNames = Array.from(cardNames);
const nodeIdx = {};
nodeNames.forEach((n, i) => { nodeIdx[n] = i; });
const N = nodeNames.length;
const adjIdx = nodeNames.map(name => {
    const nb = new Set([...(outlinks[name] || []), ...(inlinks[name] || [])]);
    return Array.from(nb).map(n => nodeIdx[n]);
});
let diameter = 0, totalPathSum = 0, pathCount = 0;
const distArr = new Int32Array(N);
for (let s = 0; s < N; s++) {
    distArr.fill(-1);
    distArr[s] = 0;
    const queue = [s];
    let head = 0, maxD = 0;
    while (head < queue.length) {
        const cur = queue[head++];
        const nd = distArr[cur] + 1;
        for (const nb of adjIdx[cur]) {
            if (distArr[nb] === -1) {
                distArr[nb] = nd;
                queue.push(nb);
                if (nd > maxD) maxD = nd;
                totalPathSum += nd;
                pathCount++;
            }
        }
    }
    if (maxD > diameter) diameter = maxD;
}
const avgPath = pathCount > 0 ? (totalPathSum / pathCount).toFixed(2) : '0';

// Top 10 枢纽（按度排序）
const hubs = Array.from(cardNames)
    .map(name => ({ name, degree: degree(name) }))
    .sort((a, b) => b.degree - a.degree)
    .slice(0, 10);

// MOC 大小
const mocFiles = dv.pages('"02-知识卡片/MOC"');
const mocSizes = {};
for (const moc of mocFiles) {
    const cardLinks = (moc.file.outlinks || []).filter(l => cardNames.has(l.path.replace(/\.md$/, '').split('/').pop()));
    mocSizes[moc.file.name] = cardLinks.length;
}
const mocsSorted = Object.entries(mocSizes).sort((a, b) => b[1] - a[1]);

// 状态分类
const ev = cards.where(p => p.status === 'evergreen').length;
const gr = cards.where(p => p.status === 'growing').length;
const sd = cards.where(p => p.status === 'seed').length;
// 时间基准
const now = new Date(); now.setHours(23,59,59,999);
const weekAgo = new Date(now.getTime() - 7*24*60*60*1000);
const recentCards = cards.where(p => p.created && new Date(p.created) >= weekAgo);
const activeDays = new Set(recentCards.map(p => String(p.created))).size;

const weekStart = new Date(now); weekStart.setDate(now.getDate() - now.getDay() + 1); weekStart.setHours(0,0,0,0);
const lastWeekStart = new Date(weekStart); lastWeekStart.setDate(weekStart.getDate() - 7);
const thisWeekCards = cards.where(p => p.created && new Date(p.created) >= weekStart).length;
const lastWeekCards = cards.where(p => p.created && new Date(p.created) >= lastWeekStart && new Date(p.created) < weekStart).length;

// 每日新增数据
const byDay = {};
for (const p of cards) {
    const dt = p.created; if (!dt) continue;
    byDay[String(dt)] = (byDay[String(dt)] || 0) + 1;
}
const dates = Object.keys(byDay).sort();
const cutoff = new Date(now.getTime() - 30*24*60*60*1000);
const dailyPts = [];
for (const d of dates) {
    if (new Date(d) < cutoff) continue;
    dailyPts.push({ date: `${new Date(d).getMonth()+1}/${new Date(d).getDate()}`, v: byDay[d] });
}

// 模块 1：知识库概览
// ═══════════════════════════════════════════════════════════
dv.header(2, '模块 1：知识库概览');

const m1 = [
  { icon:'📚', label:'卡片总数', count: cardNames.size },
  { icon:'🌲', label:'Evergreen', count: ev },
  { icon:'🌱', label:'Growing',  count: gr },
  { icon:'🌰', label:'Seed',     count: sd },
  { icon:'🔗', label:'总链接',   count: totalEdges },
];

dv.paragraph(
  '<div style="display:flex;gap:8px;margin:12px 0;flex-wrap:wrap;">' +

  '</div>'
);
```
